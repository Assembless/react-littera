// Example of using translations.

const translations = {
   example: {
     en_US: "Example",
     de_DE: "Beispiel"
   },
   hello: (name: string) => ({
     en_US: `Hello ${name}`,
     de_DE: `Hallo ${name}`
   })
}

const translated = translate(translations, "de_DE");

translated.example // => "Beispiel"
translated.hello("Mike") // => "Hallo Mike"
